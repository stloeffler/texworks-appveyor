string(TIMESTAMP DATETIME "%Y%m%d%H%M" UTC)

if (CPACK_TARGET_OS STREQUAL "win")
	set(PACKAGE_NAME "${CPACK_PACKAGE_NAME}-win")
elseif (CPACK_TARGET_OS STREQUAL "macos")
	set(PACKAGE_NAME "${CPACK_PACKAGE_NAME}-macos")
else ()
	set(PACKAGE_NAME "${CPACK_PACKAGE_NAME}")
endif ()
SET(CPACK_PACKAGE_FILE_NAME "${PACKAGE_NAME}-${CPACK_PACKAGE_VERSION}-${DATETIME}-git_${CPACK_GIT_HASH}")
